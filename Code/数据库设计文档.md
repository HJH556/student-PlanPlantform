## 1. 数据库概述

本项目采用SQLite作为关系型数据库，结合Neo4j图数据库存储知识图谱数据。本文档主要描述关系型数据库的设计。

### 1.1 数据库选型

| 数据库类型 | 用途 | 说明 |
|-----------|------|------|
| SQLite | 业务数据存储 | 用户、岗位、申请等结构化数据 |
| Neo4j | 知识图谱存储 | 职业发展路径、岗位关系等图数据 |

### 1.2 命名规范

- 表名：小写字母 + 下划线，复数形式（如：users, student_profiles）
- 字段名：小写字母 + 下划线
- 主键：id（自增整数）
- 外键：关联表名_id（如：user_id, job_id）
- 时间戳：created_at, updated_at

## 3. 建表语句

### 3.1 SQLite建表语句

```sql
-- ============================================
-- 用户表
-- ============================================
CREATE TABLE IF NOT EXISTS users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    name VARCHAR(100),
    gender VARCHAR(10),
    age INTEGER,
    phone VARCHAR(20),
    major VARCHAR(100),
    grade VARCHAR(20),
    role VARCHAR(20) DEFAULT 'user',
    avatar VARCHAR(255),
    hashed_password VARCHAR(255) NOT NULL,
    profile_step INTEGER DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- ============================================
-- 学生档案表
-- ============================================
CREATE TABLE IF NOT EXISTS student_profiles (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL UNIQUE,
    name VARCHAR(50) NOT NULL,
    major VARCHAR(100),
    grade VARCHAR(20),
    university VARCHAR(100),
    graduation_year VARCHAR(10),
    career_goal TEXT,
    current_step INTEGER DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- ============================================
-- 教育经历表
-- ============================================
CREATE TABLE IF NOT EXISTS educations (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    student_profile_id INTEGER NOT NULL,
    school VARCHAR(200) NOT NULL,
    degree VARCHAR(50),
    major VARCHAR(100),
    start_date VARCHAR(20),
    end_date VARCHAR(20),
    gpa DECIMAL(3, 2),
    courses TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (student_profile_id) REFERENCES student_profiles(id) ON DELETE CASCADE
);

-- ============================================
-- 工作/实习经历表
-- ============================================
CREATE TABLE IF NOT EXISTS experiences (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    student_profile_id INTEGER NOT NULL,
    company VARCHAR(200) NOT NULL,
    position VARCHAR(100) NOT NULL,
    start_date VARCHAR(20),
    end_date VARCHAR(20),
    description TEXT,
    achievements TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (student_profile_id) REFERENCES student_profiles(id) ON DELETE CASCADE
);

-- ============================================
-- 项目经验表
-- ============================================
CREATE TABLE IF NOT EXISTS projects (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    student_profile_id INTEGER NOT NULL,
    name VARCHAR(200) NOT NULL,
    role VARCHAR(100),
    start_date VARCHAR(20),
    end_date VARCHAR(20),
    description TEXT,
    technologies TEXT,
    achievements TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (student_profile_id) REFERENCES student_profiles(id) ON DELETE CASCADE
);

-- ============================================
-- 技能表
-- ============================================
CREATE TABLE IF NOT EXISTS skills (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    student_profile_id INTEGER NOT NULL,
    name VARCHAR(100) NOT NULL,
    level VARCHAR(20) NOT NULL DEFAULT 'beginner',
    category VARCHAR(20) NOT NULL DEFAULT 'technical',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (student_profile_id) REFERENCES student_profiles(id) ON DELETE CASCADE
);

-- ============================================
-- 证书表
-- ============================================
CREATE TABLE IF NOT EXISTS certificates (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    student_profile_id INTEGER NOT NULL,
    name VARCHAR(200) NOT NULL,
    issuer VARCHAR(200),
    issue_date VARCHAR(20),
    expiration_date VARCHAR(20),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (student_profile_id) REFERENCES student_profiles(id) ON DELETE CASCADE
);

-- ============================================
-- 获奖表
-- ============================================
CREATE TABLE IF NOT EXISTS awards (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    student_profile_id INTEGER NOT NULL,
    name VARCHAR(200) NOT NULL,
    issuer VARCHAR(200),
    date VARCHAR(20),
    description TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (student_profile_id) REFERENCES student_profiles(id) ON DELETE CASCADE
);

-- ============================================
-- 岗位表
-- ============================================
CREATE TABLE IF NOT EXISTS jobs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    title VARCHAR(200) NOT NULL,
    company VARCHAR(200) NOT NULL,
    salary VARCHAR(100),
    location VARCHAR(200),
    experience VARCHAR(100),
    education VARCHAR(100),
    description TEXT,
    requirements TEXT,
    tags VARCHAR(500),
    industry VARCHAR(200),
    type VARCHAR(100),
    publish_date VARCHAR(100),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- ============================================
-- 收藏表
-- ============================================
CREATE TABLE IF NOT EXISTS favorites (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    job_id INTEGER NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (job_id) REFERENCES jobs(id) ON DELETE CASCADE,
    UNIQUE(user_id, job_id)
);

-- ============================================
-- 申请表
-- ============================================
CREATE TABLE IF NOT EXISTS applications (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    job_id INTEGER NOT NULL,
    status VARCHAR(50) DEFAULT 'pending',
    applied_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (job_id) REFERENCES jobs(id) ON DELETE CASCADE
);

-- ============================================
-- 面试表
-- ============================================
CREATE TABLE IF NOT EXISTS interviews (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    job_id INTEGER,
    application_id INTEGER,
    title VARCHAR(200) NOT NULL,
    status VARCHAR(50) DEFAULT 'draft',
    overall_score DECIMAL(5, 2),
    overall_evaluation TEXT,
    started_at DATETIME,
    completed_at DATETIME,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (job_id) REFERENCES jobs(id) ON DELETE SET NULL,
    FOREIGN KEY (application_id) REFERENCES applications(id) ON DELETE SET NULL
);

-- ============================================
-- 面试问题表
-- ============================================
CREATE TABLE IF NOT EXISTS interview_questions (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    interview_id INTEGER NOT NULL,
    question_text TEXT NOT NULL,
    question_type VARCHAR(50) NOT NULL,
    order_index INTEGER NOT NULL,
    reference_answer TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (interview_id) REFERENCES interviews(id) ON DELETE CASCADE
);

-- ============================================
-- 面试回答表
-- ============================================
CREATE TABLE IF NOT EXISTS interview_answers (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    question_id INTEGER NOT NULL,
    user_id INTEGER NOT NULL,
    answer_text TEXT,
    score DECIMAL(5, 2),
    evaluation TEXT,
    strengths TEXT,
    weaknesses TEXT,
    suggestions TEXT,
    answered_at DATETIME,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (question_id) REFERENCES interview_questions(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- ============================================
-- 职业规划报告表
-- ============================================
CREATE TABLE IF NOT EXISTS career_reports (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    student_profile_id INTEGER,
    target_job VARCHAR(200),
    match_score DECIMAL(5, 2),
    career_path TEXT,
    recommendations TEXT,
    report_data TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (student_profile_id) REFERENCES student_profiles(id) ON DELETE SET NULL
);

-- ============================================
-- 行动计划表
-- ============================================
CREATE TABLE IF NOT EXISTS action_plans (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    career_report_id INTEGER NOT NULL,
    title VARCHAR(200) NOT NULL,
    description TEXT,
    deadline VARCHAR(20),
    priority VARCHAR(20) DEFAULT 'medium',
    status VARCHAR(20) DEFAULT 'pending',
    type VARCHAR(20) DEFAULT 'learning',
    time_range VARCHAR(20) DEFAULT 'short',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (career_report_id) REFERENCES career_reports(id) ON DELETE CASCADE
);

-- ============================================
-- 页面管理表
-- ============================================
CREATE TABLE IF NOT EXISTS pages (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR(100) NOT NULL UNIQUE,
    title VARCHAR(200) NOT NULL,
    description TEXT,
    is_enabled BOOLEAN DEFAULT 1,
    route VARCHAR(200) NOT NULL UNIQUE,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
```

## 5. 索引设计

### 5.1 主键索引

所有表的id字段都自动创建主键索引。

### 5.2 唯一索引

```sql
-- users表
CREATE UNIQUE INDEX IF NOT EXISTS idx_users_username ON users(username);
CREATE UNIQUE INDEX IF NOT EXISTS idx_users_email ON users(email);

-- student_profiles表
CREATE UNIQUE INDEX IF NOT EXISTS idx_student_profiles_user_id ON student_profiles(user_id);

-- favorites表
CREATE UNIQUE INDEX IF NOT EXISTS idx_favorites_user_job ON favorites(user_id, job_id);

-- pages表
CREATE UNIQUE INDEX IF NOT EXISTS idx_pages_name ON pages(name);
CREATE UNIQUE INDEX IF NOT EXISTS idx_pages_route ON pages(route);
```
